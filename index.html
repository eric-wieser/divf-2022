          -- name : type := value (or "term")
          def two : ℕ := 2

          def double : ℕ → ℕ :=
            λ a, two * a

          def unlikely : Prop := ∃ (x : ℕ), x*x = 2
          lemma unlikely_proof : unlikely := sorry

          lemma double_is_add_self : ∀ a, double a = a + a :=
            by {intro a, apply two_mul}
